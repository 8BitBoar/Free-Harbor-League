fhl_smugglers_road = {
    monthly_spawn_chance = 1
    
    can_start = { fhl_smugglers_road_start_trigger = yes }
    can_end = { fhl_smugglers_road_end_trigger = yes }
    
    # should be visible to countries that have the government reform and my tag (I have not switched to KOA yet)
    visible = {
        tag = FHL
    }
    
    on_start = {
        c:FHL = {
            trigger_event_non_silently = fhl_smugglers_road.777
            add_country_modifier = {
                modifier = smugglers_road_start
                years = -1
                desc = smugglers_road_start_dsec
            }
        }
    }
    
    # events! may grant temp modifiers to some locations
    on_monthly = {}
    
    # grants modifier, removes initial situation modifier
    on_ended = {
        c:FHL = {
            trigger_event_non_silently = fhl_smugglers_road_end.778
            remove_country_modifier = smugglers_road_start
            add_country_modifier = {
                name = fhl_smugglers_road_completed
                duration = -1
            }
        }
    } 

    map_color = {
        if = {
            limit = {
                owner ?= c:FHL
            }
            # custom  color later
            value = black   
        }
    }

    legend_key = {
        desc = "[pirates|e]"
        color = black
        require_color_on_map = yes
    }

}
